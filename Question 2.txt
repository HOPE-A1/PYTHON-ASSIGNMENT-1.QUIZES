inventory = {
    "Eggs": 0,
    "Milk": 0,
    "Bread": 0,
    "Sugar": 0,
    "Coffee": 0,
    "Salt": 0
}
while True:
    print("\nChoose an option:")
    print("1. Add or update item")
    print("2. Retrieve item info")
    print("3. View total quantity of all items")
    print("4. View full inventory")
    print("5. Exit")

    choice = input("Enter your choice (1-5): ")

    if choice == "1":
        item = input("Enter the item name: ").capitalize()
        if item in inventory:
            qty = int(input(f"Enter the quantity to add for {item}: "))
            inventory[item] += qty
            print(f"{qty} {item}(s) added! Total now: {inventory[item]}")
        else:
            print("Item not recognized! Allowed items: Eggs, Milk, Bread, Sugar, Coffee, Salt")

    elif choice == "2":
        item = input("Enter the item name to check: ").capitalize()
        if item in inventory:
            print(f"{item}: {inventory[item]}")
        else:
            print("Item not found in inventory!")

    elif choice == "3":
        total = sum(inventory.values())
        print(f"Total quantity of all items: {total}")

    elif choice == "4":
        print("\nCurrent Inventory:")
        for k, v in inventory.items():
            print(f"{k}: {v}")

    elif choice == "5":
        print("  Thank you for using the Inventory System! ")
        break

    else:
        print("Not available")
